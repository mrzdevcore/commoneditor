\section*{\vskip -2em  Motif}

\section*{Bilan neurologique}
\subsection*{Histoire de la maladie}
\begin{itemize}
\item [\textbf{Histoire initiale de la maladie:}]Début le 23/03/2015 par apparition de frissons  associés à une fatigue,des céphalées,une hyperthermie,une toux  grasse productive et à une douleur thoracique et à des agitations et  sursauts nocturnes.Consultation chez un médecin libre qui lui a  prescrit du Bactoclav sirop,Magnésium Calcium Biofar,ACTm et Smecta  sachet.Il y a eu amélioration de la fièvre mais après dans la  soirée vers 22h il y a eu des vomissements fait de crachat jaunâtre  puis suivi d'une perte de connaissance de 25 minutes avec révulsion  oculaire d'où hospitalisation à la clinique d'Ankadifotsy qui lui a  fait faire un EEG.
\item [\textbf{Conclusion et recommandations du dernier examen:}]	Aucun
\item [\textbf{Evolution depuis la dernière visite:}]Frissons de 15 minutes (mouvements anormaux?) le  26/03/2015 suivie d'hyperthermie,Pas de crise convulsive,bon  développement staturo-pondéral,bon développement  psychomoteur,irritable,pas de difficultés à  l'apprentissage,présente des toux grasses.
\item[\textbf{Traitement actuel et tolérance:}]DEPAKINE SB  11.36mg/kg/j en 2 prises depuis 2 jours  \\AUGMENTIN IVDL 45.45mg/kg/j en 2 prises depuis 24/03/2015  \\ROCEPHINE IVDL 45.45mg/kg/j en 2 prises depuis 24/03/2015  \\SOLUMEDROL 1.81mg/kg/j dans SGI 100cc,aérosol Salbutamol 3  ampoules,Perfalgan à la demande \\Bonne tolérance
\end{itemize}

\subsection*{Synthèse clinique}

\section*{ Conclusion et recommandations}
\begin{itemize}
